\newcommand\clr[2]{
  #1:
  \definecolor{yb#1}{HTML}{#2}
  \colorbox{yb#1}{\textcolor{white}{#2}}}
Colors:
\begin{itemize}
\item \clr{Black}{000000}
\item \clr{Red}{cf021b}
\item \clr{Blue}{053c5e}
\item \clr{Orange}{d98324}
\item \clr{Green}{31572c}
\end{itemize}

Style principles:
\begin{itemize}
\item No other colors except mentioned above and white
\item No color gradients, only solid colors
\item No color or bold in text, only italic for emphasis
\item No more than two fonts and three colors in a document
\end{itemize}
